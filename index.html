DND Stat Roller

A web GUI for a Python script I made to roll DnD stats for you. Uses PyScript.

Stat modifiers:

from pyscript import web, when from random import randint from itertools import repeat dicelist = [] statlist = [] modlist = [] stat = 0 # This function calculates one stat def roll_individual_stat(): # Rolls a 1d6 4 times for i in repeat(None, 4): dicelist.append(randint(1,6)) dicelist.sort() dicelist.pop(0) stat = sum(dicelist) statlist.append(stat) stat = 0 dicelist.clear() # Calculates ability scores for one stat (is fixed now) def calculate_stat_mod(stat): mod = round(((stat - 10)/2)) if stat == 3: mod = mod + 1 elif stat == 7: mod = mod + 1 elif stat == 11: mod = mod + 1 elif stat == 15: mod = mod + 1 elif stat == 19: mod = mod + 1 elif stat == 23: mod = mod + 1 elif stat == 27: mod = mod + 1 modlist.append(mod) # combines the previous two functions to print a list of stats and stat modifiers @when("click", "#roll-button") def roll_stats(event): for i in repeat(None, 6): roll_individual_stat() statlist_div = web.page["statlist"] statlist_div.innerText = statlist print("Stat modifiers:") for i in statlist: stat_mod = round((i - 10)/2) modlist.append(stat_mod) modlist_div = web.page["modlist"] modlist_div.innerText = modlist statlist.clear() modlist.clear()
